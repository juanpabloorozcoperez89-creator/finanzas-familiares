"""
Histórico mes a mes para detectar fugas y tendencias.
"""
import streamlit as st
import pandas as pd
import plotly.graph_objects as go
import plotly.express as px

from helpers.sheets import get_transacciones
from helpers.calc import historico_mensual


st.set_page_config(page_title="Histórico", page_icon="📈", layout="wide")
st.title("📈 Histórico mensual")
st.caption("Comparativo mes a mes: ¿estamos mejorando o empeorando?")


try:
    df = get_transacciones()
except Exception as e:
    st.error(f"Error: {e}")
    st.stop()


if df.empty:
    st.info("Aún no hay datos históricos. Empieza a registrar transacciones para ver tendencias.")
    st.stop()

n_meses = st.slider("Meses a comparar", min_value=2, max_value=12, value=6)
hist = historico_mensual(df, n_meses=n_meses)

if hist.empty:
    st.info("No hay datos suficientes para comparar.")
    st.stop()


# ============ Gráfica principal: ingresos vs egresos ============
st.subheader("Ingresos vs Egresos")

fig = go.Figure()
fig.add_trace(go.Bar(
    name="Ingresos", x=hist["año_mes_str"], y=hist["INGRESO"],
    marker_color="#2E8B57", text=hist["INGRESO"].apply(lambda v: f"Q{v:,.0f}"),
    textposition="outside",
))
fig.add_trace(go.Bar(
    name="Egresos", x=hist["año_mes_str"], y=hist["EGRESO"],
    marker_color="#C73E3A", text=hist["EGRESO"].apply(lambda v: f"Q{v:,.0f}"),
    textposition="outside",
))
fig.add_trace(go.Scatter(
    name="Diferencia", x=hist["año_mes_str"], y=hist["diferencia"],
    mode="lines+markers+text", line=dict(color="#1F4E78", width=3),
    text=hist["diferencia"].apply(lambda v: f"Q{v:,.0f}"),
    textposition="top center",
))
fig.update_layout(
    barmode="group", height=450,
    margin=dict(l=10, r=10, t=20, b=20),
    yaxis_title="Quetzales",
    hovermode="x unified",
)
st.plotly_chart(fig, use_container_width=True)

# ============ Tabla resumen ============
st.subheader("Resumen mensual")
hist_show = hist.copy()
hist_show["INGRESO"] = hist_show["INGRESO"].apply(lambda v: f"Q {v:,.2f}")
hist_show["EGRESO"] = hist_show["EGRESO"].apply(lambda v: f"Q {v:,.2f}")
hist_show["diferencia"] = hist_show["diferencia"].apply(lambda v: f"Q {v:,.2f}")
st.dataframe(
    hist_show[["año_mes_str", "INGRESO", "EGRESO", "diferencia"]].rename(columns={
        "año_mes_str": "Mes", "INGRESO": "Ingresos",
        "EGRESO": "Egresos", "diferencia": "Diferencia",
    }),
    hide_index=True,
    use_container_width=True,
)

# ============ Egresos por categoría mes a mes ============
st.divider()
st.subheader("🔥 Mapa de calor: egresos por categoría a través del tiempo")
st.caption("Identifica fugas: si una categoría aumenta mes a mes, hay un patrón ahí.")

df_eg = df[df["tipo"] == "EGRESO"].copy()
if not df_eg.empty:
    df_eg["año_mes"] = df_eg["fecha"].dt.to_period("M").astype(str)
    pivot = df_eg.pivot_table(
        index="categoria", columns="año_mes",
        values="monto", aggfunc="sum", fill_value=0,
    )
    # Solo últimos N meses
    pivot = pivot.iloc[:, -n_meses:]
    pivot = pivot.loc[pivot.sum(axis=1).sort_values(ascending=False).index]

    fig2 = px.imshow(
        pivot,
        labels=dict(x="Mes", y="Categoría", color="Q gastados"),
        aspect="auto",
        color_continuous_scale="Reds",
        text_auto=".0f",
    )
    fig2.update_layout(
        height=max(400, 25 * len(pivot) + 100),
        margin=dict(l=10, r=10, t=20, b=20),
    )
    st.plotly_chart(fig2, use_container_width=True)
else:
    st.info("No hay egresos registrados aún.")
